# startup with "NVIDIA driver ... too old" on pods with older drivers. If that hits,
# reinstall matched to the pod's driver (check `nvidia-smi` CUDA version):
#   .venvs/vllm/bin/pip install uv
#   .venvs/vllm/bin/uv pip install -U vllm --torch-backend=cu128 --python .venvs/vllm/bin/python
